for _, url in ipairs(moduleURLs) do
    local success, moduleResult = pcall(function()
        return loadstring(game:HttpGet(url))()
    end)
    if success and type(moduleResult) == "table" then
        for k, v in pairs(moduleResult) do
            BlustLib[k] = v
        end
    else
        warn("Failed to load module: " .. url .. "\nError: " .. tostring(moduleResult))
    end
end

return BlustLib
